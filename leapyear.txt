{
	a00 = 1590;
	b00 = 1760;
	while(a00 <= b00)
	{
		if(a00 % 4 == 0) 
		{
			if(a00 % 100 == 0)
			{
				if(a00 % 400 == 0)
				{
					display a00;
				}
				else
				{
					;
				}
			}
			else
			{
				display a00;
			}
		}
		else
		{
			;
		}
		a00 = a00 + 1;
	}
}
